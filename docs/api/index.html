    Quick Start

    import { Rapid, Color, Vec2 } from "rapid-render"

    // Initialize
    const rapid = new Rapid({
        canvas: document.getElementById("gameCanvas"),
        backgroundColor: Color.fromHex("E6F0FF")
    })

    // Render example
    rapid.render(() => {
        rapid.renderRect({ 
            offset: new Vec2(100, 100), 
            width: 50, 
            height: 50, 
            color: Color.Red 
        })
    })
